package io.vlingo.xoomapp.model.author;

import io.vlingo.xoom.common.version.SemanticVersion;
import io.vlingo.xoom.lattice.model.IdentifiedDomainEvent;

import java.util.*;

/**
 * See
 * <a href="https://docs.vlingo.io/xoom-lattice/entity-cqrs#commands-domain-events-and-identified-domain-events">
 *   Commands, Domain Events, and Identified Domain Events
 * </a>
 */
public final class AuthorUnrelated extends IdentifiedDomainEvent {

  public final String id;
  public final String relatedAuthor;

  public AuthorUnrelated(final String id, final String relatedAuthor) {
    super(SemanticVersion.from("1.0.0").toValue());
    this.id = id;
    this.relatedAuthor = relatedAuthor;
  }

  @Override
  public String identity() {
    return id;
  }
}
