  private World world;

  @BeforeEach
  public void setUp() {
    world = World.startWithDefaults("test-state-store-projection");
    NoOpDispatcher dispatcher = new NoOpDispatcher();
    stateStore = world.actorFor(StateStore.class, InMemoryStateStoreActor.class, Collections.singletonList(dispatcher));
    StatefulTypeRegistry statefulTypeRegistry = StatefulTypeRegistry.registerAll(world, stateStore, AuthorData.class);
    QueryModelStateStoreProvider.using(world.stage(), statefulTypeRegistry);
    entity = world.actorFor(Author.class, AuthorEntity.class, "1");
  }


  @Test
  public void withName() {
    final AuthorData firstData = AuthorData.from("26e02622-d52f-11eb-b8bc-0242ac130003", NameData.from("first-author-name-firstName", "first-author-name-lastName"), new ArrayList<>(), LocalDate.now(), new ArrayList<>(), new HashSet<>());
    final AuthorData secondData = AuthorData.from("2a4fe89c-d52f-11eb-b8bc-0242ac130003", NameData.from("second-author-name-firstName", "second-author-name-lastName"), new ArrayList<>(), LocalDate.now(), new ArrayList<>(), new HashSet<>());
    AuthorState data = firstData.toAuthorState();
    final AuthorRegistered item = entity.withName(data.name, data.availableOn).await();

    assertEquals(item.id, "26e02622-d52f-11eb-b8bc-0242ac130003");
    assertEquals(item.name.firstName, "first-author-name-firstName");
    assertEquals(item.name.lastName, "first-author-name-lastName");
  }
